加速减速

有这么个需求，类似大转盘，开始慢慢加速，从 0s 增长到 100s。也有 从 100s 减少到 0s。 可以用 setTimeout 处理

function speedDown(cb, times = 3) {
  let n = times;
  setTimeout(function start() {
    if (n === 0) {
      cb && cb();
    } else {
      console.log(`speed ${n * 100}`); // 300ms 200ms 100ms
      n--;
      setTimeout(start, n * 100);
    }
  }, n * 100);
}
